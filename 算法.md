#### 求最大公因数

gcd 欧几里得 辗转相除法

原理：a 是大数 b是小数 ，大数、小数与a mod b 拥有一样的最大公因数d（证明：a=k*c+b -> c=a/c-b/c -> c是整数 除以d左右还是整数），而d一定小于等于a mod b

```java
int gcd(int a, int b){
    return b==0 ? a : gcd(b,a%b);
}
```

#### 归并排序

```java
int n;
int temp=new int[n];
void sort(int[] nums,int lo,int hi){
    if(lo==hi) return;
    int mid=(lo+hi)/2;
    sort(nums,lo,mid);
    sort(nums,mid+1,hi);
    merge(nums,lo,mid,hi);
}
void merge(int nums[],int lo, int mid,int hi){
    int left=lo,right=mid+1;
    for(int i=lo;i<=hi;i++)
        temp[i]=nums[i];
    for(int k=lo;k<=hi;k++){
        if(left>mid) nums[k]=temp[right++];
        else if(right>hi) nums[k]=temp[left++];
        else if(temp[left]<=temp[right]) nums[k]=temp[left++];
        else if(temp[left]>temp[right])  nums[k]=temp[right++];
    }
}
```

```java
//迭代 三指针
ListNode reverse(ListNode a) {
    ListNode pre, cur, nxt;
    pre = null; cur = a; nxt = a; 
    while (cur != null) {
        // 顺序：下一个节点--前一个节点--中间节点--下一节点 Nxt放前面 避免cur更新后为null 访问cur.next越界
        nxt = cur.next;
        cur.next = pre;
        pre = cur;
        cur = nxt;
    }
    return pre;
}

//递归 神奇之处在于
//翻转之后 head的前驱节点就是head.next找到
ListNode reverse(ListNode head){
    if(head==null || head.next==null ) return head;
    ListNode last=reverse(head.next);
    head.next.next=head;
    head.next=null;
    return last;
}
```

```java
class SegTree {
    int[] tree;
    SegTree (int n) {
        this.tree = new int[4 * n];
    }
    void update(int node, int l, int r, int idx, int val) {
        if (l == r) {
            tree[node] = val;
            return;
        }
        int mid = l + (r - l) / 2, left = node * 2 + 1, right = left + 1;
        if (mid >= idx) update(left, l, mid, idx);
        else            update(right, mid + 1, r, idx);
        tree[node] = tree[left] + tree[right];
    }
    int query(int node, int l, int r, int start, int end) {
        if (l > end || r < start) return 0;
        if (start <= l && r <= end) return tree[node];
        int mid = l + (r - l) / 2, left = node * 2 + 1, right = left + 1;
        return query(left, l, mid, start, end) + query(right, mid + 1, r, start, end);
    }
}
```
